React component library

## Installation

```bash
# With npm
npm install @luminx/core
```

### Usage

```jsx
import { Button, Card, Text } from "@luminx/core";

function App() {
    return (
        <div>
            <Card withBorder>
                <Text>Hello World</Text>
                <Button>Click me</Button>
            </Card>
        </div>
    );
}
```
